rule align_reads:
	input:
		expand("aligned_reads/{sample}.sam", sample=config["samples"])

rule align_reads_wc:
	input:
		genome = "data/GENOME_PREFIX",
		r1 = "data/samples/{sample}_1.fastq",
		r2 = "data/samples/{sample}_2.fastq"
	output:
		"aligned_reads/{sample}.sam"
	shell:
		"STAR --genomeDir {input.genome} --readFilesIn {input.r1} {input.r2} ",
		"--outFileNamePrefix {wildcards.sample} --outStd SAM > {output}"
