rule align_reads:
	input:
		expand("aligned_reads/{sample}.sam", sample=config["samples"])

rule align_reads_wc:
	input:
		genome = multiext("data/GENOME_PREFIX", ".amb", ".ann", ".bwt", ".pac", ".sa"),
		r = "data/samples/{sample}.fastq"
	output:
		"aligned_reads/{sample}.sam"
	shell:
		"bwa mem data/GENOME_PREFIX {input.r} > {output}"
