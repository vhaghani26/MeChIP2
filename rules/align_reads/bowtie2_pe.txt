rule align_reads:
	input:
		expand("aligned_reads/{sample}.sam", sample=config["samples"])

rule align_reads_wc:
	input:
		genome = multiext("data/GENOME_PREFIX", ".1.bt2", ".2.bt2", ".3.bt2", ".4.bt2", ".rev.1.bt2", ".rev.2.bt2"),
		r1 = "data/samples/{sample}_1.fastq",
		r2 = "data/samples/{sample}_2.fastq"
	output:
		"aligned_reads/{sample}.sam"
	shell:
		"bowtie2 -p THREADS -x data/GENOME_PREFIX -1 {input.r1} -2 {input.r2} > {output}"
